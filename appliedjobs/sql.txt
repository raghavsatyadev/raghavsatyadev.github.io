SELECT
  "companyName",
  "externalUrl",
  "description",
  "status",
  "labels"
FROM
  jobs
WHERE
  status = 'applied'


SELECT
  "companyName",
  "externalUrl",
  "description",
  "status",
  "labels"
FROM
  jobs
WHERE
  status = 'deleted'
  AND 'Submitted' = ANY("labels");




UPDATE jobs
SET "status" = 'downloaded'
WHERE "status" = 'deleted'
  AND 'Submitted' = ANY("labels");
